UDP –ø–æ—Ö–æ–∂ –Ω–∞ –æ—Ç–ø—Ä–∞–≤–∫—É –ø–æ—á—Ç–æ–≤–æ–π –æ—Ç–∫—Ä—ã—Ç–∫–∏: –≤—ã —É–∫–∞–∑—ã–≤–∞–µ—Ç–µ –∞–¥—Ä–µ—Å –ø–æ–ª—É—á–∞—Ç–µ–ª—è (IP –∏ –ø–æ—Ä—Ç), –æ—Ç–ø—Ä–∞–≤–ª—è–µ—Ç–µ –¥–∞–Ω–Ω—ã–µ, –Ω–æ –Ω–µ –∂–¥—ë—Ç–µ –ø–æ–¥—Ç–≤–µ—Ä–∂–¥–µ–Ω–∏—è. –≠—Ç–æ –¥–µ–ª–∞–µ—Ç –µ–≥–æ –±—ã—Å—Ç—Ä—ã–º –∏ –Ω–∞–¥—ë–∂–Ω—ã–º –¥–ª—è —É–ø—Ä–∞–≤–ª–µ–Ω–∏—è —Ä–æ–±–æ—Ç–∞–º–∏.

–§–∞–π–ª `wifi_test.py`:
```python
import network
import socket
import time

# –ù–∞—Å—Ç—Ä–æ–π–∫–∏ Wi-Fi —Å–µ—Ç–∏, –∫ –∫–æ—Ç–æ—Ä–æ–π –±—É–¥–µ—Ç –ø–æ–¥–∫–ª—é—á–∞—Ç—å—Å—è ESP32
ssid = "WiFi"
password = "PASSWORD"

# –ü–æ—Ä—Ç, –Ω–∞ –∫–æ—Ç–æ—Ä–æ–º ESP32 –±—É–¥–µ—Ç –ø—Ä–∏–Ω–∏–º–∞—Ç—å –∫–æ–º–∞–Ω–¥—ã –ø–æ —Å–µ—Ç–∏
UDP_PORT = 9999

# –°–æ–∑–¥–∞—ë–º –æ–±—ä–µ–∫—Ç –¥–ª—è —É–ø—Ä–∞–≤–ª–µ–Ω–∏—è Wi-Fi –≤ —Ä–µ–∂–∏–º–µ –∫–ª–∏–µ–Ω—Ç–∞ (STA = Station)
sta = network.WLAN(network.STA_IF)
sta.active(True)  # –í–∫–ª—é—á–∞–µ–º Wi-Fi –º–æ–¥—É–ª—å

# –ü–æ–¥–∫–ª—é—á–∞–µ–º—Å—è —Ç–æ–ª—å–∫–æ –µ—Å–ª–∏ –µ—â—ë –Ω–µ –ø–æ–¥–∫–ª—é—á–µ–Ω—ã
if not sta.isconnected():
    print(f"Connecting to Wi-Fi '{ssid}'...")
    sta.connect(ssid, password)
    
    # –ñ–¥—ë–º –ø–æ–¥–∫–ª—é—á–µ–Ω–∏—è —Å —Ç–∞–π–º–∞—É—Ç–æ–º (–º–∞–∫—Å–∏–º—É–º 15 —Å–µ–∫—É–Ω–¥)
    start_time = time.time()
    while not sta.isconnected() and (time.time() - start_time) < 15:
        time.sleep(0.2)  # –¥–∞—ë–º –≤—Ä–µ–º—è Wi-Fi-—Å—Ç–µ–∫—É –∏ –Ω–µ –≥—Ä—É–∑–∏–º CPU

# –ü—Ä–æ–≤–µ—Ä—è–µ–º —Ä–µ–∑—É–ª—å—Ç–∞—Ç –ø–æ–¥–∫–ª—é—á–µ–Ω–∏—è
if sta.isconnected():
    print("Wi-Fi: connected")
    print("IP config:", sta.ifconfig())
else:
    print("Wi-Fi: FAILED to connect! Check SSID/password or router.")

# –°–æ–∑–¥–∞—ë–º UDP-—Å–æ–∫–µ—Ç (—Å–æ–∫–µ—Ç ‚Äî —ç—Ç–æ "—Ç–æ—á–∫–∞ –≤—Ö–æ–¥–∞" –¥–ª—è —Å–µ—Ç–µ–≤–æ–≥–æ –æ–±—â–µ–Ω–∏—è)
# socket.AF_INET ‚Äî –∏—Å–ø–æ–ª—å–∑—É–µ–º IPv4,
# socket.SOCK_DGRAM ‚Äî —É–∫–∞–∑—ã–≤–∞–µ—Ç, —á—Ç–æ –±—É–¥–µ—Ç –∏—Å–ø–æ–ª—å–∑–æ–≤–∞—Ç—å—Å—è –ø—Ä–æ—Ç–æ–∫–æ–ª UDP
sock = socket.socket(socket.AF_INET, socket.SOCK_DGRAM)

# –°–≤—è–∑—ã–≤–∞–µ–º (bind) —Å–æ–∫–µ—Ç —Å–æ –≤—Å–µ–º–∏ —Å–µ—Ç–µ–≤—ã–º–∏ –∏–Ω—Ç–µ—Ä—Ñ–µ–π—Å–∞–º–∏ ("0.0.0.0") –∏ –∑–∞–¥–∞–Ω–Ω—ã–º –ø–æ—Ä—Ç–æ–º
# –≠—Ç–æ –æ–∑–Ω–∞—á–∞–µ—Ç: "–ø—Ä–∏–Ω–∏–º–∞–π UDP-–ø–∞–∫–µ—Ç—ã, –ø—Ä–∏—Ö–æ–¥—è—â–∏–µ –Ω–∞ —ç—Ç–æ—Ç –ø–æ—Ä—Ç, —Å –ª—é–±–æ–≥–æ IP"
sock.bind(("0.0.0.0", UDP_PORT))

print(f"Listening on UDP port {UDP_PORT} ...")
```

### 3. UDP-—Å–µ—Ä–≤–µ—Ä —Å watchdog-–∑–∞—â–∏—Ç–æ–π

–≠—Ç–æ—Ç –∫–æ–¥ –ø—Ä–µ–≤—Ä–∞—â–∞–µ—Ç ESP32 –≤ —Å–µ—Ç–µ–≤–æ–π —Å–µ—Ä–≤–µ—Ä, –ø—Ä–∏–Ω–∏–º–∞—é—â–∏–π –∫–æ–º–∞–Ω–¥—ã –≤–∏–¥–∞:

- F,60 ‚Äî –≤–ø–µ—Ä—ë–¥ –Ω–∞ 60% —Å–∫–æ—Ä–æ—Å—Ç–∏,
- B,70 ‚Äî –Ω–∞–∑–∞–¥ –Ω–∞ 70%,
- S ‚Äî —Å—Ç–æ–ø.
–û—Å–æ–±–æ–µ –≤–Ω–∏–º–∞–Ω–∏–µ —É–¥–µ–ª–µ–Ω–æ –±–µ–∑–æ–ø–∞—Å–Ω–æ—Å—Ç–∏: –≤–≤–µ–¥—ë–Ω watchdog-—Ç–∞–π–º–µ—Ä. –ï—Å–ª–∏ –≤ —Ç–µ—á–µ–Ω–∏–µ WATCHDOG_TIMEOUT —Å–µ–∫—É–Ω–¥ –Ω–µ –ø–æ—Å—Ç—É–ø–∞–µ—Ç –Ω–∏ –æ–¥–Ω–æ–π –∫–æ–º–∞–Ω–¥—ã, –º–∞—à–∏–Ω–∫–∞ –∞–≤—Ç–æ–º–∞—Ç–∏—á–µ—Å–∫–∏ –æ—Å—Ç–∞–Ω–∞–≤–ª–∏–≤–∞–µ—Ç—Å—è. –≠—Ç–æ –ø—Ä–µ–¥–æ—Ç–≤—Ä–∞—â–∞–µ—Ç —Å–∏—Ç—É–∞—Ü–∏—é, –∫–æ–≥–¥–∞ –º–∞—à–∏–Ω–∫–∞ —É–µ–∑–∂–∞–µ—Ç ¬´–≤–¥–∞–ª—å¬ª –ø—Ä–∏ –æ–±—Ä—ã–≤–µ —Å–≤—è–∑–∏.

–§–∞–π–ª `main.py`:
```python
# main.py ‚Äî ESP32 (MicroPython)
# –£–ø—Ä–∞–≤–ª–µ–Ω–∏–µ –º–æ—Ç–æ—Ä–∞–º–∏ –ø–æ UDP: F,<pct> ‚Äî –≤–ø–µ—Ä—ë–¥, B,<pct> ‚Äî –Ω–∞–∑–∞–¥, S ‚Äî —Å—Ç–æ–ø
# –í–∫–ª—é—á—ë–Ω watchdog: –µ—Å–ª–∏ –∫–æ–º–∞–Ω–¥—ã –Ω–µ –ø—Ä–∏—Ö–æ–¥—è—Ç –¥–æ–ª—å—à–µ WATCHDOG_S ‚Äî –º–æ—Ç–æ—Ä—ã –æ—Å—Ç–∞–Ω–∞–≤–ª–∏–≤–∞—é—Ç—Å—è

import network
import socket
import time
from MX1508 import MX1508

# –ù–∞—Å—Ç—Ä–æ–π–∫–∏ Wi-Fi
ssid = "WiFi"
password = "PASSWORD"

#  –ù–∞—Å—Ç—Ä–æ–π–∫–∏ UDP 
UDP_PORT = 9999
WATCHDOG_TIMEOUT = 10  # —Å–µ–∫—É–Ω–¥—ã: –µ—Å–ª–∏ –¥–æ–ª—å—à–µ –Ω–µ—Ç –∫–æ–º–∞–Ω–¥ ‚Äî —Å—Ç–æ–ø

#  –ò–Ω–∏—Ü–∏–∞–ª–∏–∑–∞—Ü–∏—è –º–æ—Ç–æ—Ä–æ–≤ (GPIO –ø–∏–Ω—ã IN1, IN2) 
motor_L = MX1508(2, 4)    # –ª–µ–≤—ã–π –º–æ—Ç–æ—Ä
motor_R = MX1508(19, 18)  # –ø—Ä–∞–≤—ã–π –º–æ—Ç–æ—Ä

def pct_to_duty(percent):
    """–ü–µ—Ä–µ–≤–æ–¥–∏—Ç –ø—Ä–æ—Ü–µ–Ω—Ç—ã (0‚Äì100) –≤ –∑–Ω–∞—á–µ–Ω–∏–µ –®–ò–ú (0‚Äì1023) –¥–ª—è ESP32."""
    percent = max(0, min(100, int(percent)))  # –æ–≥—Ä–∞–Ω–∏—á–∏–≤–∞–µ–º –¥–∏–∞–ø–∞–∑–æ–Ω
    return int(1023 * percent / 100)

def stop():
    """–û—Å—Ç–∞–Ω–∞–≤–ª–∏–≤–∞–µ—Ç –æ–±–∞ –º–æ—Ç–æ—Ä–∞."""
    motor_L.stop()
    motor_R.stop()

def forward_pct(percent):
    """–ó–∞–ø—É—Å–∫–∞–µ—Ç –º–æ—Ç–æ—Ä—ã –≤–ø–µ—Ä—ë–¥ —Å –∑–∞–¥–∞–Ω–Ω–æ–π —Å–∫–æ—Ä–æ—Å—Ç—å—é (%)."""
    duty = pct_to_duty(percent)
    motor_L.forward(duty)
    motor_R.forward(duty)

def backward_pct(percent):
    """–ó–∞–ø—É—Å–∫–∞–µ—Ç –º–æ—Ç–æ—Ä—ã –Ω–∞–∑–∞–¥ —Å –∑–∞–¥–∞–Ω–Ω–æ–π —Å–∫–æ—Ä–æ—Å—Ç—å—é (%)."""
    duty = pct_to_duty(percent)
    motor_L.reverse(duty)
    motor_R.reverse(duty)

#  –ü–æ–¥–∫–ª—é—á–µ–Ω–∏–µ –∫ Wi-Fi 
# –°–æ–∑–¥–∞—ë–º –æ–±—ä–µ–∫—Ç –¥–ª—è —É–ø—Ä–∞–≤–ª–µ–Ω–∏—è Wi-Fi –≤ —Ä–µ–∂–∏–º–µ –∫–ª–∏–µ–Ω—Ç–∞ (STA = Station)
sta = network.WLAN(network.STA_IF)
sta.active(True)  # –í–∫–ª—é—á–∞–µ–º Wi-Fi –º–æ–¥—É–ª—å

# –ü–æ–¥–∫–ª—é—á–∞–µ–º—Å—è —Ç–æ–ª—å–∫–æ –µ—Å–ª–∏ –µ—â—ë –Ω–µ –ø–æ–¥–∫–ª—é—á–µ–Ω—ã
if not sta.isconnected():
    print(f"Connecting to Wi-Fi '{ssid}'...")
    sta.connect(ssid, password)
    
    # –ñ–¥—ë–º –ø–æ–¥–∫–ª—é—á–µ–Ω–∏—è —Å —Ç–∞–π–º–∞—É—Ç–æ–º (–º–∞–∫—Å–∏–º—É–º 15 —Å–µ–∫—É–Ω–¥)
    start_time = time.time()
    while not sta.isconnected() and (time.time() - start_time) < 15:
        time.sleep(0.2)  # –¥–∞—ë–º –≤—Ä–µ–º—è Wi-Fi-—Å—Ç–µ–∫—É –∏ –Ω–µ –≥—Ä—É–∑–∏–º CPU

# –ü—Ä–æ–≤–µ—Ä—è–µ–º —Ä–µ–∑—É–ª—å—Ç–∞—Ç –ø–æ–¥–∫–ª—é—á–µ–Ω–∏—è
if sta.isconnected():
    print("Wi-Fi: connected")
    print("IP config:", sta.ifconfig())
else:
    print("Wi-Fi: FAILED to connect! Check SSID/password or router.")

# –ë–µ–∑–æ–ø–∞—Å–Ω—ã–π —Å—Ç–∞—Ä—Ç: –º–æ—Ç–æ—Ä—ã –≤—ã–∫–ª—é—á–µ–Ω—ã
stop()

#  –ù–∞—Å—Ç—Ä–æ–π–∫–∞ UDP-—Å–µ—Ä–≤–µ—Ä–∞ 
sock = socket.socket(socket.AF_INET, socket.SOCK_DGRAM)  # UDP —Å–æ–∫–µ—Ç
sock.bind(("0.0.0.0", UDP_PORT))  # –ø—Ä–∏–Ω–∏–º–∞–µ–º —Å –ª—é–±–æ–≥–æ IP –Ω–∞ –ø–æ—Ä—Ç UDP_PORT
sock.settimeout(0.1)
print(f"UDP server listening on port {UDP_PORT}...")

last_command_time = time.time()

try:
    while True:
        #  –ü—Ä–∏—ë–º –∫–æ–º–∞–Ω–¥—ã 
        try:
            # recvfrom() –±–ª–æ–∫–∏—Ä—É–µ—Ç –≤—ã–ø–æ–ª–Ω–µ–Ω–∏–µ, –Ω–æ –º–æ–∂–Ω–æ –∑–∞–¥–∞—Ç—å —Ç–∞–π–º–∞—É—Ç —á–µ—Ä–µ–∑ settimeout(),
            data, addr = sock.recvfrom(128)
            last_command_time = time.time()
            msg = data.decode().strip()

            if not msg:
                continue

            cmd = msg[0].upper()
            speed_pct = None

            # –ü–∞—Ä—Å–∏–Ω–≥ –∫–æ–º–∞–Ω–¥—ã
            if "," in msg:
                try:
                    speed_pct = int(msg.split(",")[1])
                except ValueError:
                    pass  # –∏–≥–Ω–æ—Ä–∏—Ä—É–µ–º –Ω–µ–∫–æ—Ä—Ä–µ–∫—Ç–Ω–æ–µ —á–∏—Å–ª–æ

            # –û–±—Ä–∞–±–æ—Ç–∫–∞ –∫–æ–º–∞–Ω–¥
            if cmd == "F":  # –í–ø–µ—Ä—ë–¥
                forward_pct(speed_pct if speed_pct is not None else 70)
                sock.sendto(b"ACK\n", addr)

            elif cmd == "B":  # –ù–∞–∑–∞–¥
                backward_pct(speed_pct if speed_pct is not None else 60)
                sock.sendto(b"ACK\n", addr)

            elif cmd == "S":  # –°—Ç–æ–ø
                stop()
                sock.sendto(b"ACK\n", addr)

            elif cmd == "T":  # –ü—Ä–æ–≤–µ—Ä–∫–∞ —Å–≤—è–∑–∏
                sock.sendto(b"OK\n", addr)

        except OSError:
            pass

        #  Watchdog: –æ—Å—Ç–∞–Ω–æ–≤–∫–∞ –ø—Ä–∏ –æ—Ç—Å—É—Ç—Å—Ç–≤–∏–∏ –∫–æ–º–∞–Ω–¥ 
        if time.time() - last_command_time > WATCHDOG_TIMEOUT:
            stop()

        # –ù–µ–±–æ–ª—å—à–∞—è –ø–∞—É–∑–∞, —á—Ç–æ–±—ã –Ω–µ –≥—Ä—É–∑–∏—Ç—å –ø—Ä–æ—Ü–µ—Å—Å–æ—Ä
        time.sleep(0.01)

except KeyboardInterrupt:
    pass

finally:
    stop()
    try:
        sock.close()
    except:
        pass
    print("Bye")
```

–≤–ø–µ—Ä—ë–¥, –Ω–∞–∑–∞–¥, —Å—Ç–æ–ø, –æ—Ç–ø—Ä–∞–≤–ª—è—è –∫–æ–º–∞–Ω–¥—ã –Ω–∞ ESP32 –ø–æ UDP.

# –ü–æ—Å–ª–µ–¥–æ–≤–∞—Ç–µ–ª—å–Ω–æ—Å—Ç—å: –≤–ø–µ—Ä—ë–¥ 3 —Å–µ–∫ - –Ω–∞–∑–∞–¥ 3 —Å–µ–∫ - —Å—Ç–æ–ø

import socket
